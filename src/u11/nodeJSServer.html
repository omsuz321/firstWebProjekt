Node.js HTTP Server

This Node.js script creates a simple HTTP server that listens for requests on http://127.0.0.1:3000/. It parses the URL to get the query parameter name and sends a greeting back to the client.


        const http = require('node:http');
        const url = require('node:url');

        const hostname = '127.0.0.1';
        const port = 3000;

        const server = http.createServer((req, res) => {
        // Parses the URL to get query parameters
        const parsedUrl = url.parse(req.url, true);
        const queryName = parsedUrl.query.name || 'no input'; // Default to 'Guest' if name is not provided

        // Sets response headers
        res.setHeader('Content-Type', 'text/plain');
        res.statusCode = 200;

        // Sends a greeting back to the client using the name from the query parameters
        res.end(`Hello ${queryName}!`);
        });

        server.listen(port, hostname, () => {
        console.log(`Server running at http://${hostname}:${port}/`);
        });
